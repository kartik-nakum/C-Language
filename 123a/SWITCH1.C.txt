1.	Simple Calculator using a switch.
#include<stdio.h>
#include<conio.h>
void main()
{
  int a,b,ch;
  printf("\n enter two values");
  scanf("%d %d",&a,&b);
  printf("\n enter your choice");
  scanf("%d",&ch);
  switch(ch)
  {
  case 1:
     printf("\n addtion is %d",a+b);
     break;
     case 2:
     printf("\n subtraction is %d",a-b);
     break;
      case 3:
     printf("\n maltipication is %d",a*b);
     break;
     case 4:
     printf("\n divsion is %d",a/b);
     break;
     default:
     printf("invalid choice");

  }
  getch();
}
